import React, { useState, useMemo, useRef, useEffect } from 'react';
import { 
  Search, Camera, Loader2, Globe, MapPin, Phone, AlertCircle
} from 'lucide-react';

// ==========================================
// 🚀 外部API連携の設定
// ==========================================
// 1. スプレッドシート連携（GAS）のURL
const GAS_URL = "https://script.google.com/macros/s/AKfycbxjXsyubrqHgW_yQ3TMpu33mIsQMw-gH40ISIAPVSTa30g0AtPlFf137MhyP_tdhm9b/exec"; 

// 2. 画像解析AI（Gemini API）のキー設定
// ※この実行環境ではランタイムから自動で提供されるため空文字に設定
const apiKey = ""; 

// --- ヘルパー関数: 画像ファイルをBase64文字列に変換 ---
const fileToBase64 = (file) => {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.readAsDataURL(file);
    reader.onload = () => {
      // "data:image/png;base64,..." のカンマ以降を抽出
      const base64String = reader.result.split(',')[1];
      resolve(base64String);
    };
    reader.onerror = error => reject(error);
  });
};

// --- ヘルパー関数: API呼び出し（リトライ機能付き） ---
const fetchWithRetry = async (url, options, retries = 5) => {
  const delays = [1000, 2000, 4000, 8000, 16000];
  for (let i = 0; i < retries; i++) {
    try {
      const response = await fetch(url, options);
      if (!response.ok) {
         throw new Error(`HTTP error! status: ${response.status}`);
      }
      return await response.json();
    } catch (err) {
      if (i === retries - 1) throw err;
      await new Promise(res => setTimeout(res, delays[i]));
    }
  }
};

export default function App() {
  // LocalStorageからユーザー情報を読み込む
  const [user, setUser] = useState(() => {
    const savedUser = localStorage.getItem('gourmet_clip_user');
    return savedUser ? JSON.parse(savedUser) : null;
  });

  const [shops, setShops] = useState([]); 
  const [isLoading, setIsLoading] = useState(false); 
  const [errorMessage, setErrorMessage] = useState(""); 
  const [searchQuery, setSearchQuery] = useState("");
  const [filterVisited, setFilterVisited] = useState("all");
  
  const [isAnalyzing, setIsAnalyzing] = useState(false);
  const fileInputRef = useRef(null);

  // 初回アンケート用ステート
  const [userName, setUserName] = useState(""); 
  const [ageGroup, setAgeGroup] = useState("20代");
  const [gender, setGender] = useState("女性");

  // ==========================================
  // 📥 アプリ起動時：GASからデータ取得
  // ==========================================
  useEffect(() => {
    if (user) {
      setIsLoading(true);
      setErrorMessage("");
      
      fetch(`${GAS_URL}?userId=${user.id}`, { redirect: "follow" })
        .then(res => {
          if (!res.ok) throw new Error("ネットワークレスポンスが正常ではありません");
          return res.json();
        })
        .then(data => {
          if (data.status === 'success') {
            setShops(data.data);
          } else {
            console.error("データの取得に失敗しました", data.message);
            setErrorMessage(`データ取得エラー: ${data.message}`);
          }
        })
        .catch(err => {
          console.error("通信エラー", err);
          setErrorMessage("GASとの通信に失敗しました。");
        })
        .finally(() => setIsLoading(false));
    }
  }, [user]); 

  // ==========================================
  // 登録処理・更新処理
  // ==========================================
  const handleStart = (e) => {
    e.preventDefault();
    if (!userName.trim()) {
      alert("お名前（ニックネーム）を入力してください");
      return;
    }

    const uniqueId = "user_" + Math.random().toString(36).substring(2, 15) + Math.random().toString(36).substring(2, 15);
    const newUser = { name: userName, ageGroup, gender, id: uniqueId };
    
    localStorage.setItem('gourmet_clip_user', JSON.stringify(newUser));
    setUser(newUser);
  };

  // 📸 画像アップロード＆AI解析処理
  const handleImageUpload = async (e) => {
    const file = e.target.files[0];
    if (!file) return;

    setIsAnalyzing(true);
    setErrorMessage("");
    
    try {
      // 1. 画像をAIに送れる形式（Base64）に変換
      const base64Data = await fileToBase64(file);
      const mimeType = file.type || "image/jpeg";

      // 2. Gemini APIの設定
      const geminiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
      const promptText = `
        この画像はSNSで見つけた飲食店の紹介スクリーンショットです。
        画像内のテキストや文脈を読み取り、以下の情報を抽出してください。
        分からない項目や画像に記載がない項目は、空文字 ("") にしてください。
        
        situations（配列）には、画像の特徴から推測される利用シーンや特徴を入れてください。
        例: ["コスパ最高", "24時間営業", "デート向き", "一人飲み", "限定メニュー"]
      `;

      const payload = {
        contents: [
          {
            role: "user",
            parts: [
              { text: promptText },
              { inlineData: { mimeType: mimeType, data: base64Data } }
            ]
          }
        ],
        generationConfig: {
          responseMimeType: "application/json",
          responseSchema: {
            type: "OBJECT",
            properties: {
              prefecture: { type: "STRING", description: "都道府県（例：東京都、大阪府）" },
              area: { type: "STRING", description: "主なエリア、最寄り駅など（例：新宿、麻布十番、難波）" },
              name: { type: "STRING", description: "店舗名（正確に抽出してください）" },
              genre: { type: "STRING", description: "ジャンル（例：焼肉、スイーツ、パン、カフェなど。短く推測してください）" },
              address: { type: "STRING", description: "住所" },
              phone: { type: "STRING", description: "電話番号" },
              budget: { type: "STRING", description: "予算（例：4000~5000円）" },
              hours: { type: "STRING", description: "営業時間や定休日" }
            },
            required: ["name", "prefecture", "area", "genre"]
          }
        }
      };

      // 3. Geminiに画像を送信して解析
      const aiResponse = await fetchWithRetry(geminiUrl, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload)
      });

      // 4. AIの回答を取り出す
      const resultText = aiResponse.candidates?.[0]?.content?.parts?.[0]?.text;
      if (!resultText) throw new Error("AIから有効な回答が得られませんでした。");
      
      const parsedData = JSON.parse(resultText);
      const today = new Date();
      const dateStr = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}-${String(today.getDate()).padStart(2, '0')}`;

      // AIの結果から新しいお店データを作成
      const newShop = {
        id: Date.now(),
        userId: user.id,
        prefecture: parsedData.prefecture || "不明",
        area: parsedData.area || "不明",
        name: parsedData.name || "名称不明",
        genre: parsedData.genre || "-",
        address: parsedData.address || "",
        phone: parsedData.phone || "-",
        budget: parsedData.budget || "-",
        hours: parsedData.hours || "-",
        situations: parsedData.situations || [],
        dateAdded: dateStr,
        reservable: "-", // スプレッドシート側で手動管理
        webUrl: "#",
        visited: false,
        userMemo: ""
      };

      // 画面にすぐ追加
      setShops(prev => [newShop, ...prev]);
      
      if (fileInputRef.current) fileInputRef.current.value = "";

      // 📤 GAS（スプレッドシート）へ新規追加データを送信
      fetch(GAS_URL, {
        method: 'POST',
        headers: { "Content-Type": "text/plain" }, 
        body: JSON.stringify(newShop),
        redirect: 'follow'
      })
      .then(res => res.json())
      .then(data => console.log("シートに保存完了:", data))
      .catch(err => console.error("GAS保存エラー:", err));

    } catch (error) {
      console.error("画像解析エラー:", error);
      alert("画像の解析に失敗しました。もう一度お試しください。");
    } finally {
      setIsAnalyzing(false);
    }
  };

  const toggleVisited = (shopId) => {
    const targetShop = shops.find(s => s.id === shopId);
    if (!targetShop) return;
    
    const updatedVisited = !targetShop.visited;
    
    // 画面の表示を更新
    setShops(shops.map(shop => shop.id === shopId ? { ...shop, visited: updatedVisited } : shop));

    // 📤 GAS（スプレッドシート）へ更新データを送信
    fetch(GAS_URL, {
      method: 'POST',
      headers: { "Content-Type": "text/plain" },
      body: JSON.stringify({
        action: 'update',
        userId: user.id,
        shop: {
          id: shopId,
          visited: updatedVisited,
          userMemo: targetShop.userMemo
        }
      }),
      redirect: 'follow' // 追加
    }).catch(err => console.error("更新エラー:", err));
  };

  const updateMemo = (shopId, newMemo) => {
    // 文字を入力している最中は画面の表示だけ更新する
    setShops(shops.map(shop => shop.id === shopId ? { ...shop, userMemo: newMemo } : shop));
  };

  const saveMemoToSheet = (shopId) => {
    // 入力欄からフォーカスが外れた時（書き終わった時）にGASへ送信
    const targetShop = shops.find(s => s.id === shopId);
    if (!targetShop) return;

    fetch(GAS_URL, {
      method: 'POST',
      headers: { "Content-Type": "text/plain" },
      body: JSON.stringify({
        action: 'update',
        userId: user.id,
        shop: {
          id: shopId,
          visited: targetShop.visited,
          userMemo: targetShop.userMemo
        }
      }),
      redirect: 'follow' // 追加
    }).catch(err => console.error("メモ更新エラー:", err));
  };

  // ==========================================
  // フィルタリング・グループ化処理
  // ==========================================
  const filteredShops = useMemo(() => {
    return shops.filter(shop => {
      const matchQuery = !searchQuery || 
        shop.name.toLowerCase().includes(searchQuery.toLowerCase()) || 
        shop.area.toLowerCase().includes(searchQuery.toLowerCase()) ||
        shop.genre.toLowerCase().includes(searchQuery.toLowerCase()) ||
        shop.prefecture.toLowerCase().includes(searchQuery.toLowerCase());

      let matchVisited = true;
      if (filterVisited === "visited") matchVisited = shop.visited === true;
      if (filterVisited === "unvisited") matchVisited = !shop.visited;

      return matchQuery && matchVisited;
    });
  }, [shops, searchQuery, filterVisited]);

  const groupedShops = useMemo(() => {
    const groups = {};
    filteredShops.forEach(shop => {
      if (!groups[shop.prefecture]) groups[shop.prefecture] = {};
      if (!groups[shop.prefecture][shop.area]) groups[shop.prefecture][shop.area] = [];
      groups[shop.prefecture][shop.area].push(shop);
    });
    return groups;
  }, [filteredShops]);

  const handleReset = () => {
    if(window.confirm("データをリセットして最初の画面に戻りますか？\n（※スプレッドシートのデータは消えません）")) {
      localStorage.removeItem('gourmet_clip_user');
      setUser(null);
      setShops([]); // リセット時にリストを空にする
    }
  };


  // ==========================================
  // UI レンダー
  // ==========================================
  if (!user) {
    return (
      <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4 font-sans">
        <div className="bg-white p-8 rounded-2xl shadow-lg w-full max-w-md">
          <div className="flex justify-center mb-6">
             <div className="bg-orange-100 p-3 rounded-full text-orange-600">
               <Camera size={40} />
             </div>
          </div>
          <h1 className="text-2xl font-bold text-center text-gray-800 mb-2">GourmetClip</h1>
          <p className="text-center text-gray-500 text-sm mb-8">面倒な登録なし！あなただけのリストを作ろう</p>
          
          <form onSubmit={handleStart} className="space-y-6">
            <div>
              <label className="block text-sm font-medium text-gray-700 mb-2">お名前（ニックネーム）</label>
              <input 
                type="text"
                required
                placeholder="例：グルメ太郎"
                value={userName} 
                onChange={e => setUserName(e.target.value)}
                className="w-full border border-gray-300 rounded-xl p-3 bg-gray-50 outline-none focus:ring-2 focus:ring-orange-500"
              />
            </div>
            <div>
              <label className="block text-sm font-medium text-gray-700 mb-2">年代</label>
              <select 
                value={ageGroup} 
                onChange={e => setAgeGroup(e.target.value)}
                className="w-full border border-gray-300 rounded-xl p-3 bg-gray-50 outline-none focus:ring-2 focus:ring-orange-500"
              >
                <option>10代</option><option>20代</option><option>30代</option><option>40代</option><option>50代以上</option>
              </select>
            </div>
            <div>
              <label className="block text-sm font-medium text-gray-700 mb-2">性別</label>
              <select 
                value={gender} 
                onChange={e => setGender(e.target.value)}
                className="w-full border border-gray-300 rounded-xl p-3 bg-gray-50 outline-none focus:ring-2 focus:ring-orange-500"
              >
                <option>女性</option><option>男性</option><option>その他</option>
              </select>
            </div>
            <button type="submit" className="w-full bg-orange-600 text-white font-bold py-3 rounded-xl hover:bg-orange-700 transition-colors shadow-md">
              はじめる
            </button>
          </form>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 flex flex-col">
      <header className="bg-white sticky top-0 z-10 shadow-sm flex-shrink-0">
        <div className="p-4 flex flex-col md:flex-row md:justify-between md:items-center gap-4 border-b border-gray-100">
          <div className="flex justify-between items-center w-full md:w-auto">
            <h1 className="text-xl font-bold text-orange-600 flex items-center gap-2" onClick={handleReset} style={{cursor: 'pointer'}} title="クリックで初期化(テスト用)">
              <Camera size={24} />
              {user.name}のGourmetClip
            </h1>
            <button 
              onClick={() => fileInputRef.current?.click()}
              disabled={isAnalyzing}
              className="bg-orange-600 text-white px-4 py-2 rounded-lg text-sm font-bold flex items-center gap-2 hover:bg-orange-700 transition-colors disabled:opacity-50 md:hidden"
            >
              {isAnalyzing ? <Loader2 size={16} className="animate-spin" /> : "+ 追加"}
            </button>
          </div>

          <div className="flex items-center gap-2 w-full md:w-auto flex-1 md:max-w-xl">
            <div className="relative flex-1">
              <Search className="absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400" size={18} />
              <input 
                type="text" 
                placeholder="店名やエリアで検索..." 
                className="w-full bg-gray-100 rounded-lg py-2 pl-10 pr-4 outline-none focus:ring-2 focus:ring-orange-500 transition-all text-sm"
                value={searchQuery}
                onChange={(e) => setSearchQuery(e.target.value)}
              />
            </div>
            
            <select
              className="bg-gray-100 rounded-lg py-2 px-3 text-sm font-medium text-gray-700 outline-none focus:ring-2 focus:ring-orange-500 flex-shrink-0 cursor-pointer"
              value={filterVisited}
              onChange={(e) => setFilterVisited(e.target.value)}
            >
              <option value="all">状況: すべて</option>
              <option value="unvisited">未訪問のみ</option>
              <option value="visited">訪問済のみ</option>
            </select>

            <button 
              onClick={() => fileInputRef.current?.click()}
              disabled={isAnalyzing}
              className="hidden md:flex bg-orange-600 text-white px-4 py-2 rounded-lg text-sm font-bold items-center gap-2 hover:bg-orange-700 transition-colors whitespace-nowrap flex-shrink-0"
            >
              {isAnalyzing ? <><Loader2 size={16} className="animate-spin" /> 解析中...</> : "+ スクショ追加"}
            </button>
          </div>
          
          <input type="file" accept="image/*" className="hidden" ref={fileInputRef} onChange={handleImageUpload} />
        </div>
      </header>

      <main className="flex-1 p-4 md:p-6 overflow-hidden flex flex-col">
        {isLoading ? (
          // GASと通信中のローディング画面
          <div className="flex-1 flex flex-col items-center justify-center p-10 text-gray-400">
            <Loader2 size={32} className="animate-spin mb-4 text-orange-500" />
            <p>スプレッドシートからデータを読み込んでいます...</p>
          </div>
        ) : errorMessage ? (
          // 通信エラー時の表示
          <div className="flex-1 flex flex-col items-center justify-center p-10 text-center">
            <AlertCircle size={48} className="text-red-500 mb-4" />
            <p className="font-bold text-gray-800 mb-2">{errorMessage}</p>
            <div className="text-sm text-gray-500 bg-red-50 p-4 rounded-lg text-left inline-block mt-2">
              <p className="font-bold text-red-700 mb-1">【解決方法】</p>
              <ul className="list-disc list-inside space-y-1">
                <li>GASのデプロイ時の設定で「アクセスできるユーザー」が<strong>「全員」</strong>になっているか確認してください。</li>
                <li>設定を変更した場合は、「新しいデプロイ」を作成し、<strong>新しいURL</strong>をアプリに設定し直してください。</li>
              </ul>
            </div>
            <button 
              onClick={() => window.location.reload()}
              className="mt-6 bg-orange-100 text-orange-600 px-4 py-2 rounded-lg text-sm font-bold hover:bg-orange-200 transition-colors"
            >
              再読み込みする
            </button>
          </div>
        ) : (
          <div className="bg-white rounded-xl shadow-sm border border-gray-200 flex-1 overflow-hidden flex flex-col">
            <div className="overflow-x-auto flex-1">
              <table className="w-full text-left border-collapse min-w-[1000px]">
                <thead className="bg-gray-50 sticky top-0 z-10 shadow-sm">
                  <tr>
                    <th className="py-3 px-4 text-xs font-semibold text-gray-500 uppercase text-center w-16">訪問</th>
                    <th className="py-3 px-4 text-xs font-semibold text-gray-500 uppercase w-20">ジャンル</th>
                    <th className="py-3 px-4 text-xs font-semibold text-gray-500 uppercase w-48">店名</th>
                    <th className="py-3 px-4 text-xs font-semibold text-gray-500 uppercase w-24">営業時間</th>
                    <th className="py-3 px-4 text-xs font-semibold text-gray-500 uppercase w-28">予算</th>
                    <th className="py-3 px-4 text-xs font-semibold text-gray-500 uppercase w-24">登録日</th>
                    <th className="py-3 px-4 text-xs font-semibold text-gray-500 uppercase text-center w-20">予約</th>
                    <th className="py-3 px-4 text-xs font-semibold text-gray-500 uppercase w-48">メモ</th>
                    <th className="py-3 px-4 text-xs font-semibold text-gray-500 uppercase text-center w-16">電話</th>
                    <th className="py-3 px-4 text-xs font-semibold text-gray-500 uppercase text-center w-16">WEB</th>
                    <th className="py-3 px-4 text-xs font-semibold text-gray-500 uppercase text-center w-16">MAP</th>
                  </tr>
                </thead>
                <tbody className="divide-y divide-gray-100">
                  {Object.keys(groupedShops).length === 0 ? (
                    <tr>
                      <td colSpan="11" className="py-10 text-center text-gray-400">
                        まだお店がありません。「＋ スクショ追加」から登録してみましょう！
                      </td>
                    </tr>
                  ) : (
                    Object.entries(groupedShops).map(([prefecture, areas]) => (
                      <React.Fragment key={prefecture}>
                        <tr className="bg-gray-100">
                          <td colSpan="11" className="py-2 px-4 font-bold text-gray-800 text-sm border-y border-gray-200">
                            📍 {prefecture}
                          </td>
                        </tr>
                        {Object.entries(areas).map(([area, areaShops]) => (
                          <React.Fragment key={area}>
                            <tr className="bg-orange-50/50">
                              <td colSpan="11" className="py-1.5 px-6 font-semibold text-orange-800 text-xs border-b border-orange-100">
                                └ {area}
                              </td>
                            </tr>
                            {areaShops.map((shop) => (
                              <tr key={shop.id} className={`hover:bg-gray-50 transition-colors ${shop.visited ? 'bg-gray-50 opacity-60' : ''}`}>
                                <td className="py-3 px-4 text-center">
                                  <input 
                                    type="checkbox" 
                                    checked={shop.visited || false}
                                    onChange={() => toggleVisited(shop.id)}
                                    className="w-4 h-4 text-orange-600 rounded border-gray-300 focus:ring-orange-500 cursor-pointer"
                                  />
                                </td>
                                <td className="py-3 px-4 text-sm text-gray-600">
                                  <span className="bg-gray-100 px-2 py-1 rounded-md text-xs">{shop.genre}</span>
                                </td>
                                <td className="py-3 px-4 text-sm font-bold text-gray-900">{shop.name}</td>
                                <td className="py-3 px-4 text-xs text-gray-600">{shop.hours}</td>
                                <td className="py-3 px-4 text-xs text-gray-600">{shop.budget}</td>
                                <td className="py-3 px-4 text-xs text-gray-400">{shop.dateAdded}</td>
                                <td className="py-3 px-4 text-sm text-center font-medium text-gray-700">{shop.reservable}</td>
                                <td className="py-3 px-4">
                                  <input 
                                    type="text"
                                    placeholder="訪問日や感想を書き込めます..."
                                    value={shop.userMemo || ""}
                                    onChange={(e) => updateMemo(shop.id, e.target.value)}
                                    onBlur={() => saveMemoToSheet(shop.id)} // 入力欄から外れた時に保存
                                    className="w-full text-xs p-1.5 border border-transparent hover:border-gray-200 focus:border-orange-500 rounded bg-transparent focus:bg-white transition-all outline-none"
                                  />
                                </td>
                                <td className="py-3 px-4 text-center">
                                  <a href={shop.phone && shop.phone !== "非公開" && shop.phone !== "-" ? `tel:${shop.phone}` : "#"} 
                                     className={`inline-flex p-2 rounded-full ${shop.phone && shop.phone !== "非公開" && shop.phone !== "-" ? 'bg-green-100 text-green-600 hover:bg-green-200' : 'bg-gray-100 text-gray-300 cursor-not-allowed'}`}
                                     title={shop.phone}>
                                    <Phone size={16} />
                                  </a>
                                </td>
                                <td className="py-3 px-4 text-center">
                                  <a href={shop.webUrl && shop.webUrl !== "#" ? shop.webUrl : "#"} target="_blank" rel="noopener noreferrer"
                                     className={`inline-flex p-2 rounded-full ${shop.webUrl && shop.webUrl !== "#" ? 'bg-blue-100 text-blue-600 hover:bg-blue-200' : 'bg-gray-100 text-gray-300 cursor-not-allowed'}`} title="公式サイト・予約サイトへ">
