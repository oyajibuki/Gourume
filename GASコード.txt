// ============================================================
// GourmetClip — Google Apps Script バックエンド
// ============================================================
// 【重要】Gemini APIキーの設定方法：
//   GASエディタ → 左メニュー「プロジェクトの設定」
//   →「スクリプトプロパティ」→「プロパティを追加」
//   プロパティ名: GEMINI_API_KEY
//   値: あなたのGemini APIキー
// ============================================================

const SPREADSHEET_ID = '1uPuBBmEdSpuBxJWYTtSKnufnujPmBiJltUPNZ_4MLoo';
const SHEET_NAME = 'Shops';

// ============================================================
// doPost: 新規追加 / 更新 / AI解析
// ============================================================
function doPost(e) {
  try {
    const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(SHEET_NAME);
    const data = JSON.parse(e.postData.contents);

    // ── 0. 削除 ──
    if (data.action === 'delete') {
      const allData = sheet.getDataRange().getValues();
      const firstCell = allData.length > 0 ? allData[0][0] : '';
      const startRow = (!isNaN(Number(firstCell)) && firstCell !== '') ? 0 : 1;
      for (let i = startRow; i < allData.length; i++) {
        if (String(allData[i][0]) === String(data.shopId) && String(allData[i][1]) === String(data.userId)) {
          sheet.deleteRow(i + 1);
          return createJsonResponse({ status: 'success', message: 'deleted' });
        }
      }
      return createJsonResponse({ status: 'error', message: 'shop not found' });
    }

    // ── 1. 訪問済み・メモの更新 ──
    if (data.action === 'update') {
      const allData = sheet.getDataRange().getValues();
      for (let i = 1; i < allData.length; i++) {
        if (allData[i][0] == data.shop.id && allData[i][1] == data.userId) {
          sheet.getRange(i + 1, 16).setValue(data.shop.visited);
          sheet.getRange(i + 1, 17).setValue(data.shop.userMemo);
          return createJsonResponse({ status: 'success', message: 'updated' });
        }
      }
      return createJsonResponse({ status: 'error', message: 'shop not found' });
    }

    // ── 2. 画像解析して新規登録（OCR → Web検索 → 情報補完） ──
    if (data.action === 'analyze') {
      const apiKey = PropertiesService.getScriptProperties().getProperty('GROQ_API_KEY');
      if (!apiKey) {
        return createJsonResponse({ status: 'error', message: 'GROQ_API_KEY が未設定です。' });
      }

      const groqUrl = 'https://api.groq.com/openai/v1/chat/completions';

      // ── Step 1: OCR（Llama Vision）→ 店名・エリアだけ抽出 ──
      const ocrPayload = {
        model: 'meta-llama/llama-4-scout-17b-16e-instruct',
        messages: [{
          role: 'user',
          content: [
            {
              type: 'text',
              text: `この画像から飲食店の名前と場所を読み取ってください。
JSONのみで返してください（説明不要）：
{"name": "店名", "location": "都道府県や駅名などの手がかり"}`
            },
            { type: 'image_url', image_url: { url: `data:${data.mimeType || 'image/jpeg'};base64,${data.imageBase64}` } }
          ]
        }],
        temperature: 0.1,
        max_tokens: 256
      };

      const ocrRes = UrlFetchApp.fetch(groqUrl, {
        method: 'post',
        contentType: 'application/json',
        headers: { 'Authorization': 'Bearer ' + apiKey },
        payload: JSON.stringify(ocrPayload),
        muteHttpExceptions: true
      });

      if (ocrRes.getResponseCode() === 429) {
        return createJsonResponse({ status: 'error', isRateLimit: true, message: 'APIレート制限に達しました。1分ほど待ってから再度お試しください。' });
      }
      if (ocrRes.getResponseCode() !== 200) {
        return createJsonResponse({ status: 'error', message: `OCRエラー (HTTP ${ocrRes.getResponseCode()}): ` + ocrRes.getContentText() });
      }

      const ocrJson = JSON.parse(ocrRes.getContentText());
      let ocrText = ocrJson.choices[0].message.content.trim()
        .replace(/^```json\s*/i, '').replace(/^```\s*/i, '').replace(/\s*```$/i, '');
      const ocrMatch = ocrText.match(/\{[\s\S]*\}/);
      if (ocrMatch) ocrText = ocrMatch[0];

      let ocrInfo = { name: '', location: '' };
      try { ocrInfo = JSON.parse(ocrText); } catch(e) {}

      Logger.log('OCR結果: ' + JSON.stringify(ocrInfo));

      if (!ocrInfo.name) {
        return createJsonResponse({ status: 'error', message: '画像から店名を読み取れませんでした。別の画像をお試しください。' });
      }

      // ── Step 2: Web検索（compound-beta）→ 全カラム補完 ──
      const searchQuery = ocrInfo.name + (ocrInfo.location ? ' ' + ocrInfo.location : '') + ' 飲食店';
      const searchPayload = {
        model: 'compound-beta',
        messages: [
          {
            role: 'system',
            content: 'あなたは飲食店情報を調べるアシスタントです。必ずWeb検索して正確な情報を取得し、JSONのみで返してください。'
          },
          {
            role: 'user',
            content: `「${searchQuery}」をWeb検索して、以下のJSON形式で情報を返してください。
コードブロック・説明文は一切不要です。JSONのみ返してください。
{
  "prefecture": "都道府県（例: 東京都）",
  "area": "エリア・区・市（例: 渋谷区）",
  "name": "正式な店名",
  "genre": "ジャンル（例: 中華料理・ラーメン・焼肉）",
  "address": "住所（番地まで）",
  "phone": "電話番号",
  "budget": "予算（例: ¥1,000〜¥2,000）",
  "hours": "営業時間と定休日",
  "situations": ["おすすめシーン（デート/女子会/接待/家族など）"],
  "reservable": "予約可否（〇 △ ×）",
  "webUrl": "公式サイトURL（食べログ・ぐるなび除く）",
  "imageUrl": ""
}`
          }
        ],
        temperature: 0.1,
        max_tokens: 1024
      };

      const searchRes = UrlFetchApp.fetch(groqUrl, {
        method: 'post',
        contentType: 'application/json',
        headers: { 'Authorization': 'Bearer ' + apiKey },
        payload: JSON.stringify(searchPayload),
        muteHttpExceptions: true
      });

      if (searchRes.getResponseCode() === 429) {
        return createJsonResponse({ status: 'error', isRateLimit: true, message: 'APIレート制限に達しました。1分ほど待ってから再度お試しください。' });
      }
      if (searchRes.getResponseCode() !== 200) {
        return createJsonResponse({ status: 'error', message: `Web検索エラー (HTTP ${searchRes.getResponseCode()}): ` + searchRes.getContentText() });
      }

      const searchJson = JSON.parse(searchRes.getContentText());
      let rawText = searchJson.choices[0].message.content.trim()
        .replace(/^```json\s*/i, '').replace(/^```\s*/i, '').replace(/\s*```$/i, '');
      const jsonMatch = rawText.match(/\{[\s\S]*\}/);
      if (jsonMatch) rawText = jsonMatch[0];

      let shopInfo;
      try {
        shopInfo = JSON.parse(rawText);
      } catch (parseErr) {
        // Web検索失敗時はOCR結果をそのまま使う
        shopInfo = { name: ocrInfo.name, prefecture: '', area: ocrInfo.location || '', genre: '', address: '', phone: '', budget: '', hours: '', situations: [], reservable: '', webUrl: '', imageUrl: '' };
      }

      // 店名が空の場合はOCR結果で補完
      if (!shopInfo.name) shopInfo.name = ocrInfo.name;
      shopInfo.imageUrl = '';

      // ── 重複チェック: 同じuserId + 同じ店名が既にあるか ──
      const allRows = sheet.getDataRange().getValues();
      const shopName = shopInfo.name || '';
      const userId = data.userId || 'guest';
      let duplicateRowIndex = -1;
      let duplicateCount = 0;

      // 1行目がヘッダーかデータか判定
      const firstCell = allRows.length > 0 ? allRows[0][0] : '';
      const dataStartRow = (!isNaN(Number(firstCell)) && firstCell !== '') ? 0 : 1;

      const shopPrefecture = shopInfo.prefecture || '';
      const shopArea       = shopInfo.area || '';

      for (let i = dataStartRow; i < allRows.length; i++) {
        const rowUserId     = String(allRows[i][1]);
        const rowPrefecture = String(allRows[i][2]);
        const rowArea       = String(allRows[i][3]);
        const rowName       = String(allRows[i][4]);
        // prefecture + area + name の3つすべて一致 かつ 同じユーザー
        if (rowUserId === userId && rowPrefecture === shopPrefecture && rowArea === shopArea && rowName === shopName) {
          duplicateCount++;
          duplicateRowIndex = i;
        }
      }

      if (duplicateCount > 0) {
        // 重複あり → 既存行のメモを「xx回目の登録です」に更新してそのまま返す
        const newCount = duplicateCount + 1;
        const newMemo = `${newCount}回目の登録です`;
        sheet.getRange(duplicateRowIndex + 1, 17).setValue(newMemo); // Q列=userMemo

        const existingId = String(allRows[duplicateRowIndex][0]);
        const returnData = {
          id: existingId,
          userId: userId,
          prefecture: String(allRows[duplicateRowIndex][2]),
          area:       String(allRows[duplicateRowIndex][3]),
          name:       shopName,
          genre:      String(allRows[duplicateRowIndex][5]),
          address:    String(allRows[duplicateRowIndex][6]),
          phone:      String(allRows[duplicateRowIndex][7]),
          budget:     String(allRows[duplicateRowIndex][8]),
          hours:      String(allRows[duplicateRowIndex][9]),
          situations: String(allRows[duplicateRowIndex][10]).split(',').filter(s => s),
          dateAdded:  String(allRows[duplicateRowIndex][11]),
          reservable: String(allRows[duplicateRowIndex][12]),
          webUrl:     String(allRows[duplicateRowIndex][13]),
          imageUrl:   String(allRows[duplicateRowIndex][14]),
          visited:    allRows[duplicateRowIndex][15] === true || allRows[duplicateRowIndex][15] === 'TRUE',
          userMemo:   newMemo,
          isDuplicate: true,
          duplicateCount: newCount
        };
        return createJsonResponse({ status: 'success', data: returnData });
      }

      // ── 新規登録 ──
      const newId = new Date().getTime();
      const rowData = [
        newId,
        userId,
        shopInfo.prefecture || '',
        shopInfo.area || '',
        shopName,
        shopInfo.genre || '',
        shopInfo.address || '',
        shopInfo.phone || '',
        shopInfo.budget || '',
        shopInfo.hours || '',
        Array.isArray(shopInfo.situations) ? shopInfo.situations.join(',') : (shopInfo.situations || ''),
        data.dateAdded || new Date().toISOString().split('T')[0],
        shopInfo.reservable || '',
        shopInfo.webUrl || '',
        shopInfo.imageUrl || '',
        false,
        ''
      ];
      sheet.appendRow(rowData);

      const returnData = {
        id: newId,
        userId: userId,
        prefecture: shopInfo.prefecture || '',
        area: shopInfo.area || '',
        name: shopName,
        genre: shopInfo.genre || '',
        address: shopInfo.address || '',
        phone: shopInfo.phone || '',
        budget: shopInfo.budget || '',
        hours: shopInfo.hours || '',
        situations: Array.isArray(shopInfo.situations) ? shopInfo.situations : [],
        dateAdded: data.dateAdded || new Date().toISOString().split('T')[0],
        reservable: shopInfo.reservable || '',
        webUrl: shopInfo.webUrl || '',
        imageUrl: shopInfo.imageUrl || '',
        visited: false,
        userMemo: '',
        isDuplicate: false
      };

      return createJsonResponse({ status: 'success', data: returnData });
    }

    // ── 3. 手動入力による新規追加（action指定なし） ──
    const rowData = [
      data.id || new Date().getTime(),
      data.userId || 'guest',
      data.prefecture || '',
      data.area || '',
      data.name || '',
      data.genre || '',
      data.address || '',
      data.phone || '',
      data.budget || '',
      data.hours || '',
      Array.isArray(data.situations) ? data.situations.join(',') : (data.situations || ''),
      data.dateAdded || new Date().toISOString().split('T')[0],
      data.reservable || '',
      data.webUrl || '',
      data.imageUrl || '',
      data.visited || false,
      data.userMemo || ''
    ];
    sheet.appendRow(rowData);
    return createJsonResponse({ status: 'success', data: rowData });

  } catch (error) {
    return createJsonResponse({ status: 'error', message: error.toString() });
  }
}

// ============================================================
// doGet: ユーザーのお店一覧を取得
// ヘッダー行に依存せず固定カラム名で読み込む（ヘッダーありでも自動スキップ）
// ============================================================
function doGet(e) {
  try {
    const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(SHEET_NAME);
    const data = sheet.getDataRange().getValues();

    if (data.length === 0) {
      return createJsonResponse({ status: 'success', data: [], debug: 'シートが空です' });
    }

    // 固定カラム定義（appendRowの順番と一致）
    const COLUMNS = ['id','userId','prefecture','area','name','genre','address','phone','budget','hours','situations','dateAdded','reservable','webUrl','imageUrl','visited','userMemo'];

    // 1行目がヘッダーかデータか自動判定
    // id列（1列目）が数値ならデータ行、文字列ならヘッダー行
    const firstCellValue = data[0][0];
    const startRow = (!isNaN(Number(firstCellValue)) && firstCellValue !== '') ? 0 : 1;

    const targetUserId = e.parameter.userId;
    const result = [];
    let totalRows = 0;

    for (let i = startRow; i < data.length; i++) {
      const row = data[i];
      if (!row[0]) continue; // id空行はスキップ
      totalRows++;

      const obj = {};
      for (let j = 0; j < COLUMNS.length; j++) {
        obj[COLUMNS[j]] = (row[j] !== undefined && row[j] !== null) ? row[j] : '';
      }

      obj.id = String(obj.id); // idを文字列化
      obj.situations = obj.situations ? String(obj.situations).split(',').filter(s => s) : [];
      obj.visited = (obj.visited === true || obj.visited === 'true' || obj.visited === 'TRUE');

      if (!targetUserId || String(obj.userId) === String(targetUserId)) {
        result.push(obj);
      }
    }

    return createJsonResponse({
      status: 'success',
      data: result,
      debug: `全${totalRows}行中 ${result.length}件マッチ (userId=${targetUserId}, startRow=${startRow})`
    });

  } catch (error) {
    return createJsonResponse({ status: 'error', message: error.toString() });
  }
}

// ============================================================
// ヘルパー
// ============================================================
function createJsonResponse(responseObject) {
  return ContentService.createTextOutput(JSON.stringify(responseObject))
    .setMimeType(ContentService.MimeType.JSON);
}

  const testRes = UrlFetchApp.fetch('https://www.google.com', { muteHttpExceptions: true });

}

// ============================================================
// 【デバッグ用】Gemini APIキーと接続をテストする
// ============================================================
function testGeminiApi() {
  const apiKey = PropertiesService.getScriptProperties().getProperty('GEMINI_API_KEY');
  Logger.log('APIキー取得: ' + (apiKey ? '✅ 取得できた（' + apiKey.substring(0, 8) + '...）' : '❌ 取得できない'));
  if (!apiKey) return;

  const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${apiKey}`;
  const payload = {
    contents: [{ parts: [{ text: '「テスト成功」とだけ答えてください' }] }]
  };

  const res = UrlFetchApp.fetch(url, {
    method: 'post',
    contentType: 'application/json',
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  });
  Logger.log('HTTPステータス: ' + res.getResponseCode());
  const json = JSON.parse(res.getContentText());
  if (json.candidates) {
    Logger.log('✅ Gemini応答: ' + json.candidates[0].content.parts[0].text);
  } else {
    Logger.log('❌ エラー応答: ' + JSON.stringify(json));
  }
}
